```
#include <iostream>
#include <string>
using namespace std;

int main() {
	string name; //연두 이름
	int l = 0, o = 0, v = 0, e = 0;
	cin >> name;
	for (int i = 0; i < name.length(); i++) {//연두이름에서 love 개수 세기
		if (name[i] == 'L') l++;
		else if (name[i] == 'O') o++;
		else if (name[i] == 'V') v++;
		else if (name[i] == 'E') e++;
	}
	int num = 0;
	string team; //우승확률 높은 팀 이름
	int prob = 0; //우승확률 높은 팀의 확률
	cin >> num;
	for (int i = 0; i < num; i++) {
		string str;
		cin >> str;
		int ll = 0, oo = 0, vv = 0, ee = 0;
		for (int j = 0; j < str.length(); j++) { //입력 이름의 love 세기
			if (str[j] == 'L') ll++;
			else if (str[j] == 'O') oo++;
			else if (str[j] == 'V') vv++;
			else if (str[j] == 'E') ee++;
		}
		int val = ((ll + l + oo + o)*(ll + l + vv + v)*(ll + l + ee + e)*(oo + o + vv + v)*(oo + o + ee + e)*(vv + v + ee + e)) % 100; //확률 계산
		if (i == 0) { //처음 입력받으면 초기화해야 함
			team = str;
			prob = val;
		}
		else { //두번째 입력부터는 비교해서 저장
			if (val > prob) {
				prob = val;
				team = str;
			}
			else if (val == prob) {//확률이 같으면 사전 순 비교
				if (str < team) { // team이 사전 순으로 뒤에 나오면, 순서가 앞인 str로 바꾸기
					prob = val;
					team = str;
				}
			}
		}
	}
	cout << team << "\n";
}

#include <iostream>
#include <algorithm>
using namespace std;

int main() {
	int n;
	cin >> n;
	for (int i = 0; i < n; i++) {//test case 반복
		int s, b;
		cin >> s >> b; //세준 세비 병사 수
		int *sS = new int[s];
		for (int j = 0; j < s; j++)
			cin >> sS[j]; //세준 병사 힘
		sort(sS, sS + s); //힘 오름차순으로 정렬
		for (int j = 0; j < s; j++)
			cout << sS[j];
		cout << endl;
		int *bS = new int[b]; 
		for (int j = 0; j < b; j++)
			cin >> bS[j]; //세비 병사 힘
		sort(bS, bS + b);
		for (int j = 0; j < b; j++)
			cout << bS[j];
		cout << endl;
		int sP = 0, bP = 0; //포인터
		for (int j = 0; j < max(s, b) && sP < s && bP < b; j++) {
			if (sS[sP] < bS[bP]) //세비가 크면 세준 포인터 증가
				sP++;
			else if (sS[sP] > bS[bP]) //세준이 크면 세비 포인터 증가
				bP++;
			else if (sS[sP] == bS[bP]) //같으면 세비 포인터 증가
				bP++;
			cout << sP << bP << endl;
		}
		if (sP == s) //세준 포인터가 끝까지 갔다면 세비가 이긴 것
			cout << "B\n";
		else if (bP == b)
			cout << "S\n";
		else
			cout << "C\n";

		delete[] sS;
		delete[] bS;
	}
}

#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int main() {
	while (1) {
		int n;
		cin >> n;
		if (n == 0) break;
		string res;
		for (int i = 0; i < n; i++) {
			string str;
			cin >> str;
			if (i == 0) res = str; //res 초기화
			else { //res 갱신
				string str_tmp = str;
				string res_tmp = res;
				for (int j = 0; j < min(str.length(), res.length()); j++) {
					if (str[j] >= 'A' && str[j] <= 'Z') //대문자면 소문자로 변환
						str_tmp[j] += 32;
					if (res[j] >= 'A' && res[j] <= 'Z') //대문자면 소문자로 변환
						res_tmp[j] += 32;
					if (str_tmp[j] < res_tmp[j]) { //res가 더 늦게 나온다면
						res = str; //res 갱신
						break; //뒤는 볼 필요 없음
					}
					else if (str_tmp[j] > res_tmp[j]) { //str이 더 늦게 나온다면
						break; //res 바꾸지 않고 비교 끝내기
					}
					else if (j == min(str.length(), res.length())-1){ //j가 마지막까지 왔다는 것은, 두 문자열이 같을 경우임
						if (min(str.length(), res.length()) == str.length()) // 같을 땐 더 짧은 문자열로 res 갱신해야 함
							res = str;
					} //res 길이 == min()일 경우는, res를 갱신할 필요가 없음
				}
			}
		}
		cout << res << endl;
	}
}

## 백준 2309 일곱 난쟁이

#include <iostream>
#include <algorithm>
using namespace std;

int main() {
	int kid[9];
	int sum = 0;
	for (int z = 0; z < 9; z++) { //입력
		cin >> kid[z];
		sum += kid[z]; //키 합
	}
	sort(kid, kid + 9); //오름차순 정렬
	int i, j, flag = 0;
	for (i = 8; i > 0; i--) { //뒤에서부터 검색
		for (j = 7; j >= 0; j--) {
			if (sum - kid[i] - kid[j] == 100) { //합 - 두 키의 합 == 100 찾기
				flag = 1;
				break;
			}
		}
		if (flag) break;
	}
	for (int z = 0; z < 9; z++) {
		if (z == i || z == j) continue; // 두 키 빼고 출력
		cout << kid[z] << endl;
	}
}

## swea 1945 간단한 소인수분해

```
#include <iostream>
using namespace std;

int main() {
	int factor[5] = { 2,3,5,7,11 };
	int t;
	scanf(" %d", &t);
	for (int i = 1; i <= t; i++) {
		int n;
		scanf(" %d", &n);
        
		int res[5] = {};
		int j = 0;
		while (j < 5) {
			if (n%factor[j] == 0) { //2,3,5,7,11로 나누어 떨어지면 반복
				n /= factor[j];
				res[j]++;
			}
			else //나누어 떨어지지 않으면 다음 factor로 이동
				j++;
		}

		printf("#%d %d %d %d %d %d\n", i, res[0], res[1], res[2], res[3], res[4]);
	}
}
