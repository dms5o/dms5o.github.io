## 백준 9946 단어 퍼즐

```
#include <iostream>
#include <algorithm>
#include <string>
using namespace std;

int main() {
	int num = 0;
	while (1) {
		num++; //case 넘버
		string s1, s2;
		cin >> s1 >> s2;
		if (s1 == "END" && s2 == "END") //종료조건
			break;
		sort(s1.begin(), s1.end());
		sort(s2.begin(), s2.end());

		cout << "Case " << num << ": ";
		if (s1 == s2)
			cout  << "same" << endl;
		else
			cout << "different" << endl;
	}
}
```

## 백준 10989 수 정렬하기 3

```
#include <iostream>
using namespace std;

int main() {
	ios::sync_with_stdio(false);
	//c의 stdio와 cpp의 iostream 동기화 역할.
	//iostream, stdio 두 버퍼 모두 사용=>딜레이
	//false: cpp 독립버퍼 생성. 사용 버퍼 down=>실행속도 up
	//멀티쓰레드인 경우: 출력 순서 보장x. but 문제 풀땐 보통 싱글쓰레드 환경.
	cin.tie(NULL);
	//cin, cout 묶음 품.
	//출력 명령이나 버퍼 가득찼을 경우에만 버퍼 비우고 출력
	//입출력을 여러 번 번갈아가며 반복할 경우 필수적으로 코드 추가.
	//묶인 스트림들은 각각 IO 작업 전에 자동 버퍼 비우기 보장
	//cin은 기본적으로 먼저 출력 버퍼 비움
	int num[10001] = {};
	int cnt, tmp;
	cin >> cnt;
	for (int i = 0; i < cnt; i++) { //정렬대신 수 카운트
		cin >> tmp;
		num[tmp]++;
	}
	for (int i = 1; i < 10001; i++) {
		if (num[i]) {
			for (int j = 0; j < num[i]; j++)
				cout << i << "\n"; //endl은 개행문자 출력+출력버퍼 비우기=>딜레이
		}
	}
}
```

## 백준 1181 단어 정렬
q) 더 빠른 방법??

```
#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;

bool compare(string a, string b) {
	if (a.length() == b.length())
		return a < b;
	return a.length() < b.length();
}

int main() {
	int n;
	cin >> n;
	vector<string> word;
	for (int i = 0; i < n; i++) {//입력
		string str;
		cin >> str;
		word.push_back(str);
	}
	sort(word.begin(), word.end(), compare);

	for (int i = 0; i < n; i++) {//출력
		if (i && word[i] == word[i-1])
			continue;
		cout << word[i] << "\n";
	}
}
```
