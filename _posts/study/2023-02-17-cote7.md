## 백준 8595 히든 넘버
### 정답
데이터 타입을 항상 정확히 따져보자!  
대략 21억 선 안에서 해결될 거라 생각했는데 아니었다..  
뒤에서부터 탐색하는 방향으로 문제를 풀었는데 다른 방법도 궁금하다.  
```c++
#include <iostream>
#include <vector> //배열의 크기가 백만을 넘어가므로 동적할당 사용
#include <cmath> //제곱근 함수 사용
using namespace std;

int main() {
	int n; //n: 단어 길이
	long long sum = 0; //sum: 히든 넘버
	vector<char> str;
	scanf(" %d", &n);

	char c;
	for (int i = 0; i < n; i++) { //입력
		scanf(" %c", &c);
		str.push_back(c);
	}

	int cnt = 0; //cnt: 숫자 자릿수 체크
	for (int i = n - 1; i >= 0; i--) { //뒤에서부터 탐색
		if (str[i] >= '0' && str[i] <= '9') { //숫자면
			sum += (str[i] - '0')*pow(10, cnt); //문자를 숫자로 변환 후 저장
			cnt++; //숫자 표시
		}
		else //문자면, 자릿수 0으로 셋팅
			cnt = 0;
	}

	printf("%lld\n", sum); //출력
}
```

## 백준 1268 임시 반장 정하기
### 오답
왜 틀렸지?
```c++
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
	int n;
	scanf(" %d", &n);

	vector<vector<int>> classes(5, vector<int>(10, 0)); //2차원 벡터; 입력 저장
	vector<int> student(n, 0); //1차원 벡터; 반장 점수

	int t;
	for (int i = 0; i < n; i++) { //학생 별 입력
		for (int j = 0; j < 5; j++) { //학년 별 입력
			scanf(" %d", &t);
			classes[j][t]++;//학년 별 몇 반인지 저장
			if (classes[j][t]>=2) //2 이상; 누군가 같은 반이라면
				student[i]++; //학생 점수++
		}
	}
	int max_index = max_element(student.begin(), student.end()) - student.begin() + 1;
	printf("%d\n", max_index);
}
```

## 백준 1417 국회의원 선거
### 정답
```c++
#include <iostream>
#include <queue>
using namespace std;

int main() {
	int n;
	scanf(" %d", &n);

	priority_queue<int> pq;
	int dasom, p, cnt = 0;//dasom: 다솜 표, p: 입력받을 변수, cnt: 결과
	for (int i = 0; i < n; i++) { //입력
		scanf(" %d", &p);
		if (i > 0) //첫 번째 원소는 저장x
			pq.push(p);
		else //첫 번째 원소
			dasom = p;
	}
	int tmp; //tmp: 원소 pop해서 값 변경 후 push할 임시 변수
	while (!pq.empty()) {
		if (dasom > pq.top()) //다솜표가 가장 많으면 무한루프 종료
			break;
		tmp = pq.top();
		pq.pop();
		tmp--;
		pq.push(tmp);
		dasom++;
		cnt++;
	}
	printf("%d\n", cnt);
}
```

## 백준 2161 카드1
### 정답
```c++
#include <iostream>
#include <queue>
using namespace std;

int main() {
	int n;
	scanf(" %d", &n);

	queue<int> q;
	for (int i = 1; i <= n; i++) //1~N까지 큐에 삽입
		q.push(i);
	
	while (q.size() != 1) { //!q.empty()면 범위를 벗어남 -> 런타임 에러
		printf("%d\n", q.front()); //맨 앞 원소 출력
		q.pop(); //맨 앞 원소 제거
		q.push(q.front()); //두 번째 원소 뒤로 삽입
		q.pop(); //두 번째 원소 제거
	}
	printf("%d\n", q.front());
}
```

## SWEA 1284 수도 요금 전쟁
### 정답
```c++
#include <iostream>
using namespace std;

int main() {
	int t;
	scanf(" %d", &t);

	for (int i = 1; i <= t; i++) {//테스트케이스
		int p, q, r, s, w, rate;
		scanf(" %d %d %d %d %d", &p, &q, &r, &s, &w);
		
		int a = p * w; //a사
		int b = q; //b사
		if(w > r)
			b = (w - r)*s + q;
		
		rate = a < b ? a : b;
		printf("#%d %d\n", i, rate);
	}

	return 0;
}
```
